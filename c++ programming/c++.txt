class volume {
  public:
  void cube(int side){

  }

      cout<<"\n volume of cuoid is:"<<L*B*H;
  }

  }
  
  void sphere(int r){
      cout<<"\n volume of sphere is"<<4/3*3.14*r*r*r;
  }
};
