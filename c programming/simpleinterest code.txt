
#include <stdio.h>

    float simpleinterest(float p, float r, float t){
        
        return(p*r*t)/100;
    }

int main()
{
    float p,r,t,si;
    
    printf("\n enter the principal amount");
    scanf("%f", &p);

    printf("\n enter the rate of interest");
    scanf("%f", &r);

    printf("\n enter the time of period");
    scanf("%f", &t);
    
    si = simpleinterest(p,r,t);
    
    printf("\n the calculated simple interest is %f",si);

    return 0;
}
