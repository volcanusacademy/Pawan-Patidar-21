
#include <stdio.h>
#include <math.h>

    float campoundinterest(float p, float r, float t){
        
       float a,ci;
       a = p*((pow(1+(r/100),t)));
       ci =a-p;
       return ci;
       
    }

int main()
{
    float p,r,t,ci;
    
    printf("\n enter the principal amount");
    scanf("%f", &p);

    printf("\n enter the rate of interest");
    scanf("%f", &r);

    printf("\n enter the time of period");
    scanf("%f", &t);
    
    ci = campoundinterest(p,r,t);
    
    printf("\n the calculated simple interest is %f",ci);

    return 0;
}
