#include <stdio.h>

int main() {
    int num1, num2;
    
    printf("Enter first number: ");
    scanf("%d", &num1);

    printf("Enter second number: ");
    scanf("%d", &num2);

    
    printf("Sum: %.2d + %.2d = %.2d\n", num1, num2, num1 + num2);

    
    printf("subtraction: %.2d - %.2d = %.2d\n", num1, num2, num1 - num2);

    
    printf("multipliction: %.2d * %.2d = %.2d\n", num1, num2, num1 * num2);

    
    if (num2 != 0) {
        printf("Division: %.2d / %.2d = %.2d\n", num1, num2, num1 / num2);
    } else {
        printf("Cannot divide by zero.\n");
    }

    return 0;
}
