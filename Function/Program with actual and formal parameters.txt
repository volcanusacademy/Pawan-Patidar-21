#include <stdio.h>


float calculateVolume(float length, float width, float height) {
    return length * width * height;
}

void calculateAndPrintVolume() {
    float length, width, height, volume;

    printf("Enter the length of the cuboid: ");
    scanf("%f", &length);

    printf("Enter the width of the cuboid: ");
    scanf("%f", &width);

    printf("Enter the height of the cuboid: ");
    scanf("%f", &height);

   
    volume = calculateVolume(length, width, height);

    printf("The volume of the cuboid with length %.2f, width %.2f, and height %.2f is: %.2f\n", length, width, height, volume);
}

int main() {
    calculateAndPrintVolume();

    return 0;
}
