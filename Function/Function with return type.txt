#include <stdio.h>


int calculateSquare(int num) {
    return num * num;
}

int main() {
    int number, square;

    printf("Enter a number to find its square: ");
    scanf("%d", &number);

    
    square = calculateSquare(number);

    printf("The square of %d is: %d\n", number, square);

    return 0;
}
