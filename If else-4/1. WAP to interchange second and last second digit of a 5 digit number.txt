#include <stdio.h>

int main() {
    int num, temp;

    printf("Enter a 5-digit number: ");
    scanf("%d", &num);

    if (num < 10000 || num > 99999) {
        printf("Please enter a valid 5-digit number.\n");
        return 1; 
    }

    temp = (num % 100) / 10;
    num = num - (temp * 10);  
    num = num + (temp % 10) * 10; 

    printf("Result after interchange: %d\n", num);

    return 0; 
}
