#include <stdio.h>

int main() {
    int num, digit, sum = 0;

    printf("Enter a 4-digit number: ");
    scanf("%d", &num);

    while (num > 0) {
        digit = num % 10;
        sum += digit;
        num /= 10;
    }

    while (sum > 9) {
        int temp = sum;
        sum = 0;

        while (temp > 0) {
            digit = temp % 10;
            sum += digit;
            temp /= 10;
        }
    }

    printf("Sum of individual digits: %d\n", sum);

    return 0;
}
