#include <stdio.h>

int main() {
    int num, digit, evenCount = 0, oddCount = 0;

    printf("Enter a 5-digit number: ");
    scanf("%d", &num);

    for (int i = 0; i < 5; i++) {
        digit = num % 10;
        
        if (digit % 2 == 0) {
            evenCount++;
        } else {
            oddCount++;
        }

        num /= 10;
    }

    printf("Number of even digits: %d\n", evenCount);
    printf("Number of odd digits: %d\n", oddCount);

    return 0;
}
