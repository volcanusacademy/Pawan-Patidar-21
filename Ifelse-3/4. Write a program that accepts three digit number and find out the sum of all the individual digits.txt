#include <stdio.h>

int main() {
    int number, digit, sum = 0;

   
    printf("Enter a three-digit number: ");
    scanf("%d", &number);

    
    if (number >= 100 && number <= 999) {
        
        digit = number % 10;      
        sum += digit;
        
        digit = (number / 10) % 10;  
        sum += digit;

        digit = number / 100;       
        sum += digit;

        
        printf("The sum of individual digits is: %d\n", sum);
    } else {
        printf("Please enter a valid three-digit number.\n");
    }

    return 0;
}
