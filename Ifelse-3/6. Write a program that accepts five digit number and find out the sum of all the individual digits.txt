#include <stdio.h>

int main() {
    int number, digit, sum = 0;

    
    printf("Enter a five-digit number: ");
    scanf("%d", &number);

   
    if (number >= 10000 && number <= 99999) {
        
        digit = number % 10;        
        sum += digit;

        digit = (number / 10) % 10;    
        sum += digit;

        digit = (number / 100) % 10;  
        sum += digit;

        digit = (number / 1000) % 10;  
        sum += digit;

        digit = number / 10000;        
        sum += digit;

       
        printf("The sum of individual digits is: %d\n", sum);
    } else {
        printf("Please enter a valid five-digit number.\n");
    }

    return 0;
}
