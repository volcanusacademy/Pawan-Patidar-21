#include <stdio.h>

int main() {
    int number, originalNumber, modifiedNumber;

    
    printf("Enter a five-digit number: ");
    scanf("%d", &number);

   
    if (number >= 10000 && number <= 99999) {
        originalNumber = number;

        
        int lastDigit = number % 10;       
        int firstDigit = number / 10000;   

        
        modifiedNumber = originalNumber - (firstDigit * 10000) + (lastDigit * 10000) - lastDigit + firstDigit;

        
        printf("Modified number: %d\n", modifiedNumber);
    } else {
        printf("Please enter a valid five-digit number.\n");
    }

    return 0;
}
