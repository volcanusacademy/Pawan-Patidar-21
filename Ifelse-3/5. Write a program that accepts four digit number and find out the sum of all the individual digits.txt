#include <stdio.h>

int main() {
    int number, digit, sum = 0;

   
    printf("Enter a four-digit number: ");
    scanf("%d", &number);

    
    if (number >= 1000 && number <= 9999) {
       
        digit = number % 10;         
        sum += digit;

        digit = (number / 10) % 10;    
        sum += digit;

        digit = (number / 100) % 10;  
        sum += digit;

        digit = number / 1000;         
        sum += digit;

        
        printf("The sum of individual digits is: %d\n", sum);
    } else {
        printf("Please enter a valid four-digit number.\n");
    }

    return 0;
}
