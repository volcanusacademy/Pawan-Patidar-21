#include <stdio.h>

int main() {
    char rentCar, carType, haveCoupon;

    
    printf("Are you renting a car? (Y/N): ");
    scanf(" %c", &rentCar);

    if (rentCar == 'Y' || rentCar == 'y') {
       
        printf("Enter 'C' for Compact or 'F' for Full-size: ");
        scanf(" %c", &carType);

        if (carType == 'C' || carType == 'c') {
           
            printf("Do you have a coupon? (Y/N): ");
            scanf(" %c", &haveCoupon);

            if (haveCoupon == 'Y' || haveCoupon == 'y') {
               
                printf("You have selected Compact. Price with 7%% off: %.2f\n", 456.56 * 0.93);
            } else {
               
                printf("You have selected Compact. Normal price: 456.56\n");
            }
        } else if (carType == 'F' || carType == 'f') {
           
            printf("Do you have a coupon? (Y/N): ");
            scanf(" %c", &haveCoupon);

            if (haveCoupon == 'Y' || haveCoupon == 'y') {
               
                printf("You have selected Full-size. Price with 7%% off: %.2f\n", 460.50 * 0.93);
            } else {
               
                printf("You have selected Full-size. Normal price: 460.50\n");
            }
        } else {
            printf("Invalid car type input.\n");
        }
    } else if (rentCar == 'N' || rentCar == 'n') {
        printf("Thank you for using our service. Have a great day!\n");
    } else {
        printf("Invalid input for renting a car.\n");
    }

    return 0;
}
