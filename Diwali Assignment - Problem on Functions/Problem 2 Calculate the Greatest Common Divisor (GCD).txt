#include <stdio.h>


int calculateGCD(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int main() {
    int num1, num2;

    
    printf("Enter the first number: ");
    scanf("%d", &num1);

    printf("Enter the second number: ");
    scanf("%d", &num2);

    
    if (num1 < 0 || num2 < 0) {
        printf("Invalid input. Please enter non-negative integers.\n");
        return 1;  
    }

    int gcd = calculateGCD(num1, num2);
    printf("Greatest Common Divisor (GCD) of %d and %d is: %d\n", num1, num2, gcd);

    return 0;  
}
