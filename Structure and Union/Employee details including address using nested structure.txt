#include <stdio.h>


struct Address {
    char area[50];
    int pinCode;
    char street[50];
};


struct Employee {
    char name[50];
    int employeeId;
    struct Address empAddress;
};

int main() {
    struct Employee emp;

   
    printf("Enter employee name: ");
    scanf("%s", emp.name);

    printf("Enter employee ID: ");
    scanf("%d", &emp.employeeId);

    
    printf("Enter employee's area: ");
    scanf("%s", emp.empAddress.area);

    printf("Enter pin code: ");
    scanf("%d", &emp.empAddress.pinCode);

    printf("Enter employee's street: ");
    scanf("%s", emp.empAddress.street);

    
    printf("\nEmployee Details:\n");
    printf("Name: %s\n", emp.name);
    printf("Employee ID: %d\n", emp.employeeId);
    printf("Address:\n");
    printf("Area: %s\n", emp.empAddress.area);
    printf("Pin Code: %d\n", emp.empAddress.pinCode);
    printf("Street: %s\n", emp.empAddress.street);

    return 0;
}
