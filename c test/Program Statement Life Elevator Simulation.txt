#include <stdio.h>
#include <unistd.h>

int currentfloor = 0;

void gotofloor(int newfloor) {
    if (newfloor > 0 && newfloor <= 10) {
        if (newfloor > currentfloor) {
            printf("Going up...\n");
            while (currentfloor < newfloor) {
                sleep(1);
                printf("Current Floor: %d\n", ++currentfloor);
            }
        } else if (newfloor < currentfloor) {
            printf("Going down...\n");
            while (currentfloor > newfloor) {
                sleep(1);
                printf("Current Floor: %d\n", --currentfloor);
            }
        }
        printf("Reached Floor %d\n", currentfloor);
    } else {
        printf("Invalid floor request\n");
    }
}

int main() {
    int destination;
    char choice = 'y';
    while (choice == 'y' || choice == 'Y') {
        printf("Enter the floor you want to go (1-10): ");
        scanf("%d", &destination);
        gotofloor(destination);
        printf("Do you want to go to another floor (y/n): ");
        getchar(); 
        scanf(" %c", &choice); 
    }
    printf("Exiting Elevator\n");
    return 0;
}
