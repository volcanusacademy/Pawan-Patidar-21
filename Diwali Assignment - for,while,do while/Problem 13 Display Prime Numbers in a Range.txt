#include <stdio.h>


int isPrime(int num) {
    if (num < 2) {
        return 0;  
    }

    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return 0;  
        }
    }

    return 1;  
}

int main() {
    int start, end;

   
    printf("Enter the start of the range: ");
    scanf("%d", &start);

    printf("Enter the end of the range: ");
    scanf("%d", &end);

    
    if (start > end) {
        printf("Invalid range. Start should be less than or equal to end.\n");
        return 1; 
    }

   
    printf("Prime numbers in the range [%d, %d]:\n", start, end);
    for (int i = start; i <= end; i++) {
        if (isPrime(i)) {
            printf("%d\n", i);
        }
    }

    return 0;
}
