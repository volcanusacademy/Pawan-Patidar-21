#include <stdio.h>


int isPerfect(int num) {
    int sum = 0;
    
    
    for (int i = 1; i <= num / 2; ++i) {
        if (num % i == 0) {
            sum += i;
        }
    }

   
    return (sum == num);
}

int main() {
    int start, end;

   
    printf("Enter the start of the range: ");
    scanf("%d", &start);

    printf("Enter the end of the range: ");
    scanf("%d", &end);

    
    printf("Perfect numbers in the range %d to %d are:\n", start, end);
    for (int i = start; i <= end; ++i) {
        if (isPerfect(i)) {
            printf("%d\n", i);
        }
    }

    return 0;
}
