#include <stdio.h>

int main() {
    int n;

   
    printf("Enter a positive integer: ");
    scanf("%d", &n);

    
    if (n <= 0) {
        printf("Invalid input. Please enter a positive integer.\n");
        return 1; 
    }

   
    printf("Hailstone sequence for %d: ", n);

    while (n != 1) {
        printf("%d -> ", n);

        if (n % 2 == 0) {
           
            n = n / 2;
        } else {
           
            n = 3 * n + 1;
        }
    }

   
    printf("%d\n", n);

    return 0;
}
