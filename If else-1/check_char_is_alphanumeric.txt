//c program to check enter character is $ or & .

#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar == '$' || inputChar == '@') {
        printf("The entered character '%c' is either '$' or '@'.\n", inputChar);
    } else {
        printf("The entered character '%c' is not '$' or '@'.\n", inputChar);
    }

    return 0;
}
