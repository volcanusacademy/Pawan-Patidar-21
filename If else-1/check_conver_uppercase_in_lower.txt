//c program to check char is lower case or not ,if not so convert in lower case.


#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);
    
    if (inputChar >= 'A' && inputChar <= 'Z') {
        inputChar = inputChar + ('a' - 'A');
        printf("The entered character was in uppercase. Converted to lowercase: %c\n", inputChar);
    } else {
        printf("The entered character '%c' is not in uppercase.\n", inputChar);
    }

    return 0;
}
