// c program to check which number is greater.

#include <stdio.h>

int main() {
    int num1, num2;

    printf("Enter the first number: ");
    scanf("%d", &num1);

    printf("Enter the second number: ");
    scanf("%d", &num2);

    if (num1 == num2) {
        printf("Both numbers are equal.\n");
    } else {
        int greatestNumber = (num1 > num2) ? num1 : num2;
        printf("The greatest number is: %d\n", greatestNumber);
    }

    return 0;
}
