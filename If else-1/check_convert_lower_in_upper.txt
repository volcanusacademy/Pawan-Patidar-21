//c program if charr is in upper or not is not convert in upper case.

#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar >= 'a' && inputChar <= 'z') {
        inputChar = inputChar - ('a' - 'A');
        printf("The entered character was in lowercase. Converted to uppercase: %c\n", inputChar);
    } else {
        printf("The entered character '%c' is not in lowercase.\n", inputChar);
    }

    return 0;
}
