//c program to check word is in uppercase or not.

#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar >= 'A' && inputChar <= 'Z') {
        printf("The entered character '%c' is in uppercase.\n", inputChar);
    } else {
        printf("The entered character '%c' is not in uppercase.\n", inputChar);
    }

    return 0;
}
