//c program to check enter char is digit or not.

#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar >= '0' && inputChar <= '9') {
        printf("The entered character '%c' is a digit.\n", inputChar);
    } else {
        printf("The entered character '%c' is not a digit.\n", inputChar);
    }

    return 0;
}
