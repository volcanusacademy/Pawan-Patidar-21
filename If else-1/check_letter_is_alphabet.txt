//c program to check enter character is alphabet or not.

#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if ((inputChar >= 'A' && inputChar <= 'Z') || (inputChar >= 'a' && inputChar <= 'z')) {
        printf("The entered character '%c' is an alphabet.\n", inputChar);
    } else {
        printf("The entered character '%c' is not an alphabet.\n", inputChar);
    }

    return 0;
}
