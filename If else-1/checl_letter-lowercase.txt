// c program to check word is in lower case or not.

#include <stdio.h>

int main() {
    char inputChar;
    
    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar >= 'a' && inputChar <= 'z') {
        printf("The entered character '%c' is in lowercase.\n", inputChar);
    } else {
        printf("The entered character '%c' is not in lowercase.\n", inputChar);
    }

    return 0;
}
