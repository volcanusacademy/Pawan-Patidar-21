#include <stdio.h>

int main() {
    char inputChar;

    printf("Enter a character: ");
    scanf(" %c", &inputChar);

    if (inputChar >= '0' && inputChar <= '9') {
        inputChar = '*';
        printf("The entered character was a digit. Replacement: %c\n", inputChar);
    } else {
        printf("The entered character '%c' is not a digit.\n", inputChar);
    }

    return 0;
}
