

#include <stdio.h>

int main()
{
    int num, counter=0;
    
    printf("\n enter your number");
    scanf("%d",&num);
    
    for(int i=1;i<num;i++){
    if(num%i==0){
        counter++;
    }
        
    }
    if(counter>1){
        
        printf("\n number is not prime");
    }
    else{
        
        printf("\n number is prime");
    }
    

    return 0;
}
