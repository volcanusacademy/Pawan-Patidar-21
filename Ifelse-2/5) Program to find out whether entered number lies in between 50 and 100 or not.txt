#include <stdio.h>

int main() {
    int number;

    printf("Enter an number: ");
    scanf("%d", &number);

    if (number >= 50 && number <= 100) {
        printf("%d lies between 50 and 100.\n", number);
    } else {
        printf("%d does not lie between 50 and 100.\n", number);
    }

    return 0;
}
