#include <stdio.h>

int main() {
    char ch;

    printf("Enter your choice : ");
    scanf(" %c", &ch);  

    if ((ch >= '!' && ch <= '/') || (ch >= ':' && ch <= '@') ||
        (ch >= '[' && ch <= '`') || (ch >= '{' && ch <= '~')) {
        printf("%c is a special symbol.\n", ch);
    } else {
        printf("%c is not a special symbol.\n", ch);
    }

    return 0;
}
