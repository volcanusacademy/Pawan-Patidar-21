#include <stdio.h>

int main() {
    int marks[5];
    float percentage;

    printf("Enter marks for 5 subjects:\n");
    for (int i = 0; i < 5; ++i) {
        printf("Enter marks for subject %d: ", i + 1);
        scanf("%d", &marks[i]);
    }

    percentage = (float)(marks[0] + marks[1] + marks[2] + marks[3] + marks[4]) / 5;

    if (percentage >= 75 && percentage <= 100) {
        printf("Grade A\n");
    } else if (percentage >= 60 && percentage < 75) {
        printf("Grade B\n");
    } else if (percentage >= 50 && percentage < 60) {
        printf("Grade C\n");
    } else if (percentage >= 33 && percentage < 50) {
        printf("Grade D\n");
    } else {
        printf("Fail\n");
    }

    return 0;
}
