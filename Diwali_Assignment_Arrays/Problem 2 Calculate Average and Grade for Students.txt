#include <stdio.h>

int main() {
    int n;

   
    printf("Enter the number of students: ");
    scanf("%d", &n);

   
    int scores[n];

    
    printf("Enter the scores:\n");
    for (int i = 0; i < n; ++i) {
        printf("Score for student %d: ", i + 1);
        scanf("%d", &scores[i]);
    }

   
    int sum = 0;
    for (int i = 0; i < n; ++i) {
        sum += scores[i];
    }
    float average = (float)sum / n;

   
    printf("\nResults:\n");
    printf("Average Score: %.2f\n", average);

    printf("Grades:\n");
    for (int i = 0; i < n; ++i) {
        printf("Student %d: ", i + 1);

       
        if (scores[i] >= 90) {
            printf("A\n");
        } else if (scores[i] >= 80) {
            printf("B\n");
        } else if (scores[i] >= 70) {
            printf("C\n");
        } else if (scores[i] >= 60) {
            printf("D\n");
        } else {
            printf("F\n");
        }
    }

    return 0;
}
