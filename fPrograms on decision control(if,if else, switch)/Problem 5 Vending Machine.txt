#include <stdio.h>

int main() {
   
    float productPrice, amountInserted;

    
    printf("Enter the price of the product: $");
    scanf("%f", &productPrice);

    printf("Enter the amount of money inserted: $");
    scanf("%f", &amountInserted);

    
    if (amountInserted >= productPrice) {
       
        float change = amountInserted - productPrice;

      
        printf("Purchase successful! Your change is $%.2f.\n", change);
    } else {
        
        printf("Insufficient funds. Please insert more money.\n");
    }

    return 0;
}
