#include <stdio.h>

void MultiplicationTable(int n);

int main() {
    int number;

    printf("Enter an integer: ");
    scanf("%d", &number);

    MultiplicationTable(number);

    return 0; 
}

void MultiplicationTable(int n) {
    printf("Multiplication Table for %d:\n", n);

    for (int i = 1; i <= 10; i++) {
        printf("%d x %d = %d\n", n, i, n * i);
    }
}
