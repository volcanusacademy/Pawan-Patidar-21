#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int generateRandomNumber() {
   
    srand(time(NULL));

    
    int randomNum = rand() % 100 + 1;

    return randomNum;
}

int main() {
   
    int randomNumber = generateRandomNumber();

    printf("Random number between 1 and 100: %d\n", randomNumber);

    return 0;
}
