#include <stdio.h>

float calculateAverage(int num1, int num2) {
   
    float average = (num1 + num2) / 2.0;
    
    return average;
}

int main() {
   
    int number1 = 30;
    int number2 = 40;
    
    float result = calculateAverage(number1, number2);
    
    printf("The average of %d and %d is: %.2f\n", number1, number2, result);

    return 0;
}
