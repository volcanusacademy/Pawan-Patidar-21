#include <stdio.h>

float calculateAverage(int num1, int num2);

int main() {
    int num1, num2;

    printf("Enter two integers: ");
    scanf("%d %d", &num1, &num2);

    float average = calculateAverage(num1, num2);

    printf("Average: %.2f\n", average);

    return 0; 
}

float calculateAverage(int num1, int num2) {

    float average = (float)(num1 + num2) / 2;

    return average;
}
